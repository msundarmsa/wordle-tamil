<script>

    $(document).ready(function(){
        $(document).keypress(function(event) {
                var keycode = (event.keyCode ? event.keyCode : event.which);
                if (keycode == "116") {
                        var selection = window.getSelection().toString();
			var q_param = $.trim(selection);
			q_param = q_param.replace(/[;]/g, '%3B');
			var q_param = q_param.replace(/\/?#/g,'')
			//console.log(selection)
			//console.log(q_param)
			var q_url = 'tamil/get_transliteration.py?q=' + q_param;
			$.ajax({
				url:q_url,
				dataType: "text"
			}).done(function(data){
				//console.log(data)
				var title_text = selection.replace(/\u00a0\u00a0/g,'\u00a0') 
				title_text = title_text.replace(/\u00a0/g,'<br/>')
				var title = '<span id="transliteration_modal_override">Transliteration for: ' + title_text + '</span>';
				var display_data = '<span id="transliteration_modal_override">' + data + '</span>';
				$('#tranliterationModal').find('.modal-title').html(title);
				$('#tranliterationModal').find('.modal-body').html(display_data);
				$('#tranliterationModal').modal('show');
			}).fail(function(xhr, ajaxOptions, thrownError) {
				console.log(xhr)
				console.log(ajaxOptions)
				console.log(thrownError)
				if (thrownError === 'Request-URI Too Long') {
					var message = "Please select a smaller passage to transliterate";
				}
				else if (thrownError === 'Internal Server Error') {
					var message = "The transliteration code encountered a problem. Please try again.";
				}
				var title = "Error: " + thrownError;
				//var message = "Please select a smaller passage to transliterate";
				$('#tranliterationModal').find('.modal-title').text(title);
				$('#tranliterationModal').find('.modal-body').text(message);
				$('#tranliterationModal').modal('show');
			});
                };
        });
    });

</script>
